async function guardarPublicacion() { if (!window.sqlClient) { alert("No hay conexión con la base de datos Neon."); return; } const titulo = document.getElementById('in-titulo').value.trim(); const sector = document.getElementById('in-sector').value.trim(); const descripcion = document.getElementById('in-desc').value.trim(); const fileInputFotos = document.getElementById('in-file-foto').files; const fileVideo = document.getElementById('in-file-video').files[0]; if (!titulo || !descripcion) { alert("Por favor ingrese al menos el Título y la Descripción del evento."); return; } // Validación de tamaño para el video (Límite sugerido: 6 MB para servidores Serverless) if (fileVideo && fileVideo.size > 6 * 1024 * 1024) { alert("⚠️ El video es demasiado pesado (" + (fileVideo.size / (1024*1024)).toFixed(1) + " MB).\n\nPara que la plataforma funcione de forma rápida y sin errores en Neon, por favor sube un video corto de menos de 6 MB o comprimido."); return; } try { let fotosUrls = []; const hiddenFotoVal = document.getElementById('in-url-foto-hidden').value; if (fileInputFotos && fileInputFotos.length > 0) { for (let i = 0; i < fileInputFotos.length; i++) { const base64 = await convertirBase64(fileInputFotos[i]); fotosUrls.push(base64); } } else if (hiddenFotoVal) { try { fotosUrls = JSON.parse(hiddenFotoVal); if (!Array.isArray(fotosUrls)) fotosUrls = [hiddenFotoVal]; } catch(e) { fotosUrls = [hiddenFotoVal]; } } const url_foto_json = JSON.stringify(fotosUrls); let url_video = document.getElementById('in-url-video-hidden').value; if (fileVideo) { url_video = await convertirBase64(fileVideo); } const fechaHoy = new Date().toISOString().split('T')[0]; if (editId) { await window.sqlClient` UPDATE bitacora_eventos SET titulo = ${titulo}, sector = ${sector}, descripcion = ${descripcion}, url_foto = ${url_foto_json}, url_video = ${url_video} WHERE id = ${Number(editId)} `; } else { await window.sqlClient` INSERT INTO bitacora_eventos (titulo, sector, descripcion, url_foto, url_video, fecha) VALUES (${titulo}, ${sector}, ${descripcion}, ${url_foto_json}, ${url_video}, ${fechaHoy}) `; } cerrarModal(); cargarBitacora(); } catch (err) { alert("Error al procesar los archivos o guardar en Neon:\n" + err.message); } }
